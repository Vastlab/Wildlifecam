337 void Update_Threshold_Background(int operationMode, Mat inputFrame,
338  Mat Background1, Mat Background2, Mat Thresholds, Mat Blobs,
339  Mat classificationMap, int GLOBAL_THRESHOLD) {
340 
341  //If operationMode:
342  // 0 = first frame or re-initializing
343  // 1 = Normal operation
344  // 2 = Lighting or Motion Change
345  // 3 = Global Increment successful
346  // 4 = Global Increment not successful
347 
348 
349  //Threshold Initialization: The initial threshold is initialized by taking the MAX PIXEL VALUE-MIN PIXEL
350  //VALUE in a 3x3 pixel grid of the current image frame.
351 
352  //Background Initialization: Both backgrounds are also initialized in the threshold initialization block where
353 
354  //Background1= MIN PIXEL VALUE in a 3x3 pixel grid with the current pixel at the center
355  //Background2= MAX PIXEL VALUE in a 3x3 pixel grid with the current pixel at the center
356 
357  double maxValue;
358  double minValue;
359  double max_min_Value;
360 
361  int Process_Counter = 0; //updated when a pixel is updated
362 
363  int scale = inputFrame.rows / Blobs.rows;
364 
365  int alphaFactor = 0; //this gets passed to blending
366 
367  if (operationMode == 0) { // (means first frame or reinitialization)
368  //Initialize the background and thresholds
369  for (int i = 1; i < inputFrame.rows - 1; i++)
370  for (int j = 1; j < inputFrame.cols - 1; j++) {
371  cv::Rect myROI(j - 1, i - 1, 2, 2);
372 
373  Mat crop(inputFrame, myROI); // NOTE: this will only give you a reference to the ROI of the original data
374 
375  //find the max/min and max-min value
376  cv::minMaxLoc(crop, &minValue, &maxValue); //find minimum and maximum intensities
377  max_min_Value = maxValue - minValue;
378 
379  //assign values to the backgrounds and threshold
380  Background1.at<uchar>(i, j) = (uchar) minValue;
381  Background2.at<uchar>(i, j) = (uchar) maxValue;
382  Thresholds.at<uchar>(i, j) = (uchar) max_min_Value;
383  }
384 
385  } //if operationMode = 0
386  else {
387  FU_Counter++;
388  //Initialize the background and thresholds
389  for (int i = 0; i < inputFrame.rows; i++)
390  for (int j = 0; j < inputFrame.cols; j++) {
391  Process_Counter++;
392  Process_Counter %= UPDATE_NUMBER;
393 
394  if (Process_Counter == FU_Counter) {
395 
396  if(operationMode == 1) {
397  //Update Background using Incremental Model
398  Update_Background_Incremental(inputFrame, Background1, i, j);
399  }
400  else if(operationMode == 2) { // lighting or motion
401  //Update Background using Incremental Model
402  Update_Background_Incremental(inputFrame, Background1, i, j);
403 
404  //global threshold also incremented by GLOBAL_INCREMENT
405  GLOBAL_THRESHOLD += GLOBAL_INCREMENT;
406  }
407  else if(operationMode == 3) { // global inc successful
408  //Alpha Blending model level 1: non-target
409  Update_Background_Blend1(inputFrame, Background1, Background2, alphaFactor, i, j);
410  }
411  else if(operationMode == 4) { // global inc fail
412  //Alpha Blending model level 2: all
413  Update_Background_Blend2(inputFrame, Background1, i, j);
414  }
415 
416  //Update Thresholds - always the same
417  Update_Threshold(classificationMap,Thresholds, Blobs, i, j, scale, minValue);
418 
419 
420  } //process = fu
421 
422  }
423  } // not first frame
424 
425 }

444 void Background_Subtraction(Mat inputFrame, Mat Background1, Mat Background2,
445  Mat Thresholds, Mat Detected_Image_HI, Mat Detected_Image_LOW,
446  int GLOBAL_THRESHOLD, int HI_THRESHOLD_INC) {

448  Mat diffImage = cv::Mat::zeros(inputFrame.rows, inputFrame.cols, CV_8UC1);

450  //do the business
451  uchar abs_b1, abs_b2, temp_b1, temp_b2;
452 
453  for (int i = 1; i < inputFrame.rows - 1; i++)
454  for (int j = 1; j < inputFrame.cols - 1; j++) {
455 
456  abs_b1 = abs(
457  inputFrame.at<uchar>(i, j) - Background1.at<uchar>(i, j));
458  abs_b2 = abs(
459  inputFrame.at<uchar>(i, j) - Background2.at<uchar>(i, j));
460 
461  if (abs_b2 < abs_b1) {
462  //switch the background pixels
463  temp_b1 = Background1.at<uchar>(i, j);
464  temp_b2 = Background2.at<uchar>(i, j);
465 
466  Background1.at<uchar>(i, j) = temp_b2;
467  Background2.at<uchar>(i, j) = temp_b1;
468 
469  diffImage.at<uchar>(i, j) = abs_b1;
470  }
471 
472  else {
473  diffImage.at<uchar>(i, j) = abs_b2;
474 
475  }
476 
477  //threshold HI
478  if (diffImage.at<uchar>(i, j)
479  >= (Thresholds.at<uchar>(i, j) + (uchar) GLOBAL_THRESHOLD

482  Detected_Image_LOW.at<uchar>(i, j) = 0;
483  }
484  //threshold LOW
485  else if (diffImage.at<uchar>(i, j)
486  > Thresholds.at<uchar>(i, j) + GLOBAL_THRESHOLD) {
487  Detected_Image_HI.at<uchar>(i, j) = 0;
488  Detected_Image_LOW.at<uchar>(i, j) = 255;
489  }
490  //not detected
491  else {
492  Detected_Image_LOW.at<uchar>(i, j) = 0;
493  Detected_Image_HI.at<uchar>(i, j) = 0;
494  }

600  std::vector<cv::Point2i> blob;
601 
602  for (int i = rect.y; i < (rect.y + rect.height); i++) {

605  continue;
606  }
607 
608  blob.push_back(cv::Point2i(j, i));
609  }
610  }
