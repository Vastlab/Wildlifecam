331 /* weibull_cdf computes the probability (given our assumptions) that the value x is an outlier ABOVE the fit distribution. if the distribution was non-match data, then it provides this probability that x is a match score. If data was match-data then it would be the probability of it being a larger non-match.
332 computes @f[ 1-e^{{x/scale}^{shape}} @f]
333 
334 @param x the location at which to compute the probability of being an outlier
335 @param scale the scale parameter of the Weibull. This is the first element in weibullparms (as computed by our weibull_fit)
336 @param shape the scale parameter of the Weibull. This is the first second in weibullparms (as computed by our weibull_fit)
337 @return if in the range [0-1] it is the probability of X being an outlier. Any value < 0 is an error code. returns -1 for invalid scale <=0 , -2 for invalid shape <=0
338 *
339 */
340 double weibull_cdf(double x, double scale, double shape)
341 {
342  double cdf;
343  double tempVal, tempVal1;
344 
345  if(x<0) return 0; // optimize for the simple case that can be common in playing with SVMs. (a valid value, and can ignore other possible errors)
346 
347 #ifndef WEIBULL_IGNORE_ERRORS
348 #ifdef WEIBULL_USE_ASSERTS
349  assert(scale>0);
350  assert(shape>0);
351 #else
352  if(scale<=0) WEIBULL_ERROR_HANDLER( -1, "Bad scale in weibull_cdf");
353  if(shape <=0) WEIBULL_ERROR_HANDLER(-2, "Bad shape in weibull_cdf");
354 #endif
355 #endif
356 
357 
358

361  cdf = 1-exp(-1*tempVal1);

511  del = (y-z)*(y+z);

534  result=tempVal/tempVal1;

543 #ifdef _WIN32

545 #else
546  test = isfinite(result);
547 #endif

549  result = 0;

556  }

558  {

570  y = exp(z*z) * exp(del);
571  result = (y+y) - result;
572  }
573  }

585 static double derfcinv(double x)
586 {
587 
588  double a [] = {1.370600482778535e-02, -3.051415712357203e-01, 1.524304069216834, -3.057303267970988, 2.710410832036097, -8.862269264526915e-01};
589  double b [] = {-5.319931523264068e-02, 6.311946752267222e-01, -2.432796560310728, 4.175081992982483, -3.320170388221430};
590  double c [] = {5.504751339936943e-03, 2.279687217114118e-01, 1.697592457770869, 1.802933168781950, -3.093354679843504, -2.077595676404383};
591  double d [] = {7.784695709041462e-03, 3.224671290700398e-01, 2.445134137142996, 3.754408661907416};
592  double q,r,u;
593 
594  double result = 0;
595  double xlow = 0.0485000000;
596  double xhigh = 1.9515000000;
597  double tosp = 1.1283791670955126645;
598  double xk;
599  double tempVal, tempVal1, tempVal2;
600 
601  xk = x;
602 
603  //Rational approximation for central region
604  if ((xlow <= xk) && (xk <= xhigh))
605  {
606  q = xk - 1;
607  r = q*q;
608 
609  tempVal1=(((((a[0]*r+a[1])*r+a[2])*r+a[3])*r+a[4])*r+a[5])*q;
610  tempVal2=((((b[0]*r+b[1])*r+b[2])*r+b[3])*r+b[4])*r+1;
611 
612  result = tempVal1/tempVal2;
613  }
614 
615  //Rational approximation for lower region
616  else if ((0 < xk) && (xk < xlow))
617  {
618  tempVal=xk/2;
619  q = sqrt(-2*log(tempVal));
620 
621  tempVal1=((((c[0]*q+c[1])*q+c[2])*q+c[3])*q+c[4])*q+c[5];
622  tempVal2=(((d[0]*q+d[1])*q+d[2])*q+d[3])*q+1;
623 
624  result=tempVal1/tempVal2;
625  }
626 
627  //Rational approximation for upper region
628  else if ((xhigh < xk) && (xk < 2))
629  {
630  tempVal=xk/2;
631  q = sqrt(-2*log(1-tempVal));
632  tempVal1= -1*(((((c[0]*q+c[1])*q+c[2]))*q+c[3])*q+c[4])*q+c[5];
633  tempVal2= (((d[0]*q+d[1])*q+d[2])*q+d[3])*q+1;
634 
635  result = tempVal1/tempVal2;
636 
637  }
638 
639  // Root finding with Halley's method
640  // For f = erfc(x) - y so then f' = -2/sqrt(pi)*exp(-x^2), and f" = -2*x*f'
641 
642 
643  double erfcx = wcalcerfc(result);
644  if(erfcx <0) WEIBULL_ERROR_HANDLER(-9999,"derfcinv fails because wcalcerfc failed");
645  tempVal= (-1*tosp);
646  tempVal1=(-1*result)*(result);
647  tempVal1=exp(tempVal1);
648  tempVal2=tempVal*tempVal1;
649 
650  u = (erfcx - xk)/tempVal2;
651 
652  tempVal=(1+result*u);
653 
654  tempVal1= u / tempVal;
655 
656  result = result - tempVal1;
657 
658  return result;
659 
660 }

663 static int weibull_neg_log_likelihood(double* nlogL, double* acov, double* weibulparms, double* data,
664  double* censoring, double* frequency, int size)
665 {
666  int i;
667  double mu = weibulparms[0]; /* scale */
668  double sigma = weibulparms[1]; /* shape */
669 
670  if (sigma <= 0) WEIBULL_ERROR_HANDLER(-1,"Bad sigma (shape) in weibull_neg_log_likelihood..");
671 
672  double* z = (double*)malloc(sizeof(double)*size);
673  double* expz = (double*)malloc(sizeof(double)*size);
674  double* L = (double*)malloc(sizeof(double)*size);
675  double logSigma;
676 
677  logSigma = log(sigma);
678 
679  for (i = 0; i < size; i++)
680  {
681  z[i] = (data[i]-mu)/sigma;
682  expz[i] = exp(z[i]);
683  L[i] = (z[i]-logSigma)*(1-censoring[i]-expz[i]);
684  }
685 
686  // Sum up the individual contributions, and return the negative log-likelihood.
687  for (i=0; i<size; i++)
688  {
689  *nlogL += (frequency[i]*L[i]);
690  }
691 
692  *nlogL = *nlogL * -1;
693 
694  //Compute the negative hessian at the parameter values, and invert to get
695  //the observed information matrix.
696  double* unc=(double*)malloc(sizeof(double)*size);
697  double nH11=0;
698  double nH12=0;
699  double nH22=0;
700 
701  for (i=0; i<size; i++)
702  {
703  unc[i]=(1-censoring[i]);
704  nH11=nH11+(frequency[i]*expz[i]);
705  }
706 
707  for (i=0; i<size; i++)
708  {
709  nH12=nH12+(frequency[i] * ((z[i] + 1) * expz[i] - unc[i]));
710  nH22=nH22+(frequency[i] * (z[i] *(z[i] + 2) * expz[i] - ((2 * z[i] + 1) *unc[i])));
711  }
712 
713  double sigmaSq = sigma * sigma;
714  double avarDenom = (nH11*nH22 - nH12*nH12);
715 
716  acov[0]=sigmaSq*(nH22/avarDenom);
717  acov[1]=sigmaSq*((-1*nH12)/avarDenom);
718  acov[2]=sigmaSq*((-1*nH12)/avarDenom);
719  acov[3]=sigmaSq*(nH11/avarDenom);
720 
721  free(z);
722  free(expz);
723  free(L);
724  free(unc);
725  return 0;
726 }

759 // based on dfzero from fortan, it finxs the zero in the given search bands, and stops if it is within tolerance.
760 static int wdfzero(double* sigmahat, double* likelihood_value, double* err, double* search_bands, double tol,
761  double* x0, double* frequency, double meanUncensored, int size)
762 {
763  double exitflag;
764  double a,b,c=0.0,d=0.0,e=0.0,m,p,q,r,s;
765  double fa,fb,fc;
766  double fval;
767  double tolerance;
768 
769  exitflag=1;
770  *err = exitflag;
771 
772  a = search_bands[0];
773  b = search_bands[1];
774 
775  fa = weibull_scale_likelihood(a,x0,frequency,meanUncensored,size);
776  fb = weibull_scale_likelihood(b,x0,frequency,meanUncensored,size);
777 
778  if (fa == 0)
779  {
780  b=a;
781  *sigmahat=b;
782  fval = fa;
783  *likelihood_value = fval;
784  return 1;
785  }
786  else if (fb == 0)
787  {
788  // b = b;
789  fval=fb;
790  *likelihood_value = fval;
791  *sigmahat=b;
792  return 1;
793  }
794  else if ((fa > 0) == (fb > 0))
795  {
796  WEIBULL_ERROR_HANDLER(-4,"ERROR: wdfzero says function values at the interval endpoints must differ in sign\n");
797  }
798 
799  fc = fb;
800 
801  //Main loop, exit from middle of the loop
802  while (fb != 0)
803  {
804  // Insure that b is the best result so far, a is the previous
805  // value of b, and c is on the opposite of the zero from b.
806  if ((fb > 0) == (fc > 0))
807  {
808  c = a;
809  fc = fa;
810  d = b - a;
811  e = d;
812  }
813 
814  double absFC;
815  double absFB;
816 
817  absFC=fabs(fc);
818  absFB=fabs(fb);
819 
820  if (absFC < absFB)
821  {
822  a = b;
823  b = c;
824  c = a;
825  fa = fb;
826  fb = fc;
827  fc = fa;
828  }
829 
830  //set up for test of Convergence, is the interval small enough?
831  m = 0.5*(c - b);
832 
833  double absB, absM, absFA, absE;
834  absB=fabs(b);
835  absM=fabs(m);
836  absFA=fabs(fa);
837  absFB=fabs(fb);
838  absE=fabs(e);
839 
840  double maxB = (absB > 1.0) ? absB : 1.0;
841  tolerance = 2.0*tol*maxB;
842 
843  if ((absM <= tolerance) | (fb == 0.0))
844  break;
845 
846  //Choose bisection or interpolation
847  if ((absE < tolerance) | (absFA <= absFB))
848  {
849  //Bisection
850  d = m;
851  e = m;
852  }
853  else
854  {
855  //Interpolation
856  s = fb/fa;
857 
858  if (a == c)
859  {
860  //Linear interpolation
861  p = 2.0*m*s;
862  q = 1.0 - s;
863  }
864  else
865  {
866  //Inverse quadratic interpolation
867  q = fa/fc;
868  r = fb/fc;
869  p = s*(2.0*m*q*(q - r) - (b - a)*(r - 1.0));
870  q = (q - 1.0)*(r - 1.0)*(s - 1.0);
871  }
872 
873  if (p > 0)
874  q = -1.0*q;
875  else
876  p = -1.0*p;
877 
878  double tempTolerance = tolerance*q;
879  double absToleranceQ;// = (tempTolerance < 0) ? -1.0 * tempTolerance : tempTolerance;
880  absToleranceQ=fabs(tempTolerance);
881  double tempEQ = (0.5 * e * q);
882  double absEQ;// = (tempEQ < 0) ? -1.0 * tempEQ : tempEQ;
883  absEQ=fabs(tempEQ);
884 
885  //Is interpolated point acceptable
886  if ((2.0*p < 3.0*m*q - absToleranceQ) & (p < absEQ))
887  {
888  e = d;
889  d = p/q;
890  }
891  else
892  {
893  d = m;
894  e = m;
895  }
896 
897  } //Interpolation
898 
899  //Next point
900  a = b;
901  fa = fb;
902 
903  double absD=fabs(d);
904  if (absD > tolerance)
905  b = b + d;
906  else if (b > c)
907  b = b - tolerance;
908  else
909  b = b + tolerance;
910 
911  fb = weibull_scale_likelihood(b,x0,frequency,meanUncensored,size);
912 
913  }//Main loop (While)
914 
915  fval=weibull_scale_likelihood(b,x0,frequency,meanUncensored,size);
916  *likelihood_value = fval;
917  *sigmahat=b;
918 
919  return 1;
920 }
